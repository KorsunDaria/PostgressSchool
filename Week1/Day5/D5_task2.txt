#include "postgres.h"
#include "tcop/dest.h"

typedef struct
{
	DestReceiver pub;		
	int64		count;		
} CountDestReceiver;


static bool
countReceiveSlot(TupleTableSlot *slot, DestReceiver *self)
{
	CountDestReceiver *myState = (CountDestReceiver *) self;

	myState->count++;
	return true;
}

static void
countStartupReceiver(DestReceiver *self, int operation, TupleDesc typeinfo)
{
	CountDestReceiver *myState = (CountDestReceiver *) self;

	myState->count = 0;
	elog(LOG, "count_receiver: начало приёма кортежей");
}

static void
countShutdownReceiver(DestReceiver *self)
{
	CountDestReceiver *myState = (CountDestReceiver *) self;

	elog(LOG, "count_receiver: обработано строк: " INT64_FORMAT,
		 myState->count);
}

static void
countDestroyReceiver(DestReceiver *self)
{
	pfree(self);
}

DestReceiver *
CreateCountDestReceiver(void)
{
	CountDestReceiver *self =
		(CountDestReceiver *) palloc0(sizeof(CountDestReceiver));

	self->pub.receiveSlot = countReceiveSlot;
	self->pub.rStartup = countStartupReceiver;
	self->pub.rShutdown = countShutdownReceiver;
	self->pub.rDestroy = countDestroyReceiver;
	self->pub.mydest = MyDest;

	self->count = 0;

	return (DestReceiver *) self;
}
